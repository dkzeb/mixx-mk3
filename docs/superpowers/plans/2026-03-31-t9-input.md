```python
"""T9 multi-tap text input engine.

Stateless with respect to hardware — receives pad indices, produces text events
via callbacks. No HID, no xdotool, no LED logic.
"""
import time

# Pad-to-character map. Keys are physical pad numbers (1-16).
# Character pads cycle through their list on repeated presses.
CHAR_MAP = {
    13: [" "],
    14: ["a", "b", "c", "2"],
    15: ["d", "e", "f", "3"],
    9:  ["g", "h", "i", "4"],
    10: ["j", "k", "l", "5"],
    11: ["m", "n", "o", "6"],
    5:  ["p", "q", "r", "s", "7"],
    6:  ["t", "u", "v", "8"],
    7:  ["w", "x", "y", "z", "9"],
    2:  ["0", "+"],
    1:  ["#"],
    3:  ["*"],
}

PAD_ENTER = 4
PAD_CANCEL = 12
PAD_BACKSPACE = 16

COMMIT_TIMEOUT = 0.8  # seconds


class T9Engine:
    def __init__(self, on_change=None, on_submit=None, on_cancel=None):
        self.on_change = on_change
        self.on_submit = on_submit
        self.on_cancel = on_cancel
        self._buffer = []           # committed characters
        self._pending_pad = None    # pad currently cycling (or None)
        self._pending_index = 0     # index into CHAR_MAP[pad]
        self._last_press_time = 0.0

    def press(self, pad):
        """Handle a pad press. Returns the current display text."""
        now = time.monotonic()

        if pad == PAD_ENTER:
            self._commit_pending()
            text = self.get_text()
            if self.on_submit:
                self.on_submit(text)
            return text

        if pad == PAD_CANCEL:
            self._pending_pad = None
            self._pending_index = 0
            self._buffer.clear()
            if self.on_cancel:
                self.on_cancel()
            return ""

        if pad == PAD_BACKSPACE:
            if self._pending_pad is not None:
                # Cancel the uncommitted character
                self._pending_pad = None
                self._pending_index = 0
            elif self._buffer:
                self._buffer.pop()
            text = self.get_text()
            if self.on_change:
                self.on_change(text)
            return text

        chars = CHAR_MAP.get(pad)
        if chars is None:
            return self.get_text()

        if pad == self._pending_pad and (now - self._last_press_time) < COMMIT_TIMEOUT:
            # Same pad within timeout — cycle to next character
            self._pending_index = (self._pending_index + 1) % len(chars)
        else:
            # Different pad or timeout expired — commit pending, start new
            self._commit_pending()
            self._pending_pad = pad
            self._pending_index = 0

        self._last_press_time = now
        text = self.get_text()
        if self.on_change:
            self.on_change(text)
        return text

    def tick(self):
        """Call from main loop. Commits pending character if timeout elapsed.
        Returns True if a commit happened."""
        if self._pending_pad is None:
            return False
        if (time.monotonic() - self._last_press_time) >= COMMIT_TIMEOUT:
            self._commit_pending()
            if self.on_change:
                self.on_change(self.get_text())
            return True
        return False

    def get_text(self):
        """Return current text: committed buffer + pending character."""
        text = "".join(self._buffer)
        if self._pending_pad is not None:
            chars = CHAR_MAP.get(self._pending_pad, [])
            if chars:
                text += chars[self._pending_index]
        return text

    def get_pending_pad(self):
        """Return the pad currently cycling, or None."""
        return self._pending_pad

    def reset(self):
        """Clear all state for a new input session."""
        self._buffer.clear()
        self._pending_pad = None
        self._pending_index = 0
        self._last_press_time = 0.0

    def _commit_pending(self):
        """Move pending character into the committed buffer."""
        if self._pending_pad is not None:
            chars = CHAR_MAP.get(self._pending_pad, [])
            if chars:
                self._buffer.append(chars[self._pending_index])
            self._pending_pad = None
            self._pending_index = 0
```

```python
#!/usr/bin/env python3
"""T9 multi-tap input daemon for the MK3.

Reads HID reports from hidraw, runs the T9 multi-tap engine,
controls pad LEDs, and injects keystrokes via xdotool.

Runs as a systemd service alongside Mixxx.
"""
import os
import sys
import glob
import time
import struct
import subprocess

from mk3_t9_engine import (
    T9Engine, CHAR_MAP, PAD_ENTER, PAD_CANCEL, PAD_BACKSPACE, COMMIT_TIMEOUT,
)

# --------------------------------------------------------------------------
# MK3 USB / HID constants
# --------------------------------------------------------------------------
VENDOR_ID = "17CC"
PRODUCT_ID = "1600"

# Keyboard button: byte 0x04, mask 0x02 in Report 0x01
KEYBOARD_BYTE = 0x04
KEYBOARD_MASK = 0x02

# Pad report
PAD_REPORT_ID = 0x02
PAD_DATA_START = 1
PAD_ENTRY_LENGTH = 3
PAD_MAX_ENTRIES = 21
PAD_COUNT = 16
PAD_PRESSURE_THRESHOLD = 256

# HW index to physical pad number (same as JS mapping)
PAD_HW_TO_PHYSICAL = [13, 14, 15, 16, 9, 10, 11, 12, 5, 6, 7, 8, 1, 2, 3, 4]

# Physical pad number to LED byte offset in Report 0x81
# p1=byte26 ... p16=byte41 in HW order (p1=top-left, p16=bottom-right)
# Physical-to-HW: physical 13->p1, 14->p2, ..., 1->p13, 2->p14, ...
PAD_PHYSICAL_TO_LED_BYTE = {
    1: 38, 2: 39, 3: 40, 4: 41,     # bottom row -> p13,p14,p15,p16
    5: 34, 6: 35, 7: 36, 8: 37,     # row 2 -> p9,p10,p11,p12
    9: 30, 10: 31, 11: 32, 12: 33,  # row 3 -> p5,p6,p7,p8
    13: 26, 14: 27, 15: 28, 16: 29, # top row -> p1,p2,p3,p4
}

# Indexed color palette (same values as MaschineMK3.Color in JS)
COLOR_OFF = 0
COLOR_RED = 4
COLOR_GREEN = 20
COLOR_CYAN = 32
COLOR_BLUE = 40
COLOR_WHITE = 68

# LED colors for T9 mode
COLOR_CHAR_IDLE = COLOR_CYAN      # character pads when idle
COLOR_CHAR_ACTIVE = COLOR_WHITE   # pad currently cycling
COLOR_ENTER = COLOR_GREEN         # enter pad
COLOR_CANCEL = COLOR_RED          # cancel pad
COLOR_BACKSPACE = 32              # backspace (dim white — mono brightness in indexed space)

# Character pads = all pads in CHAR_MAP
CHAR_PADS = set(CHAR_MAP.keys())
SPECIAL_PADS = {PAD_ENTER, PAD_CANCEL, PAD_BACKSPACE}
ALL_T9_PADS = CHAR_PADS | SPECIAL_PADS


# --------------------------------------------------------------------------
# HID device helpers
# --------------------------------------------------------------------------
def find_mk3_hidraw():
    """Find the hidraw device for the MK3 HID interface."""
    for hidraw in sorted(glob.glob("/dev/hidraw*")):
        try:
            devnum = hidraw.replace("/dev/hidraw", "")
            uevent_path = f"/sys/class/hidraw/hidraw{devnum}/device/uevent"
            if os.path.exists(uevent_path):
                with open(uevent_path) as f:
                    content = f.read().upper()
                    if VENDOR_ID in content and PRODUCT_ID in content:
                        return hidraw
        except (IOError, OSError):
            continue
    return None


def open_hidraw_read(path):
    """Open hidraw for reading."""
    return os.open(path, os.O_RDONLY)


def open_hidraw_write(path):
    """Open hidraw for writing (LED output)."""
    return os.open(path, os.O_WRONLY)


# --------------------------------------------------------------------------
# LED output
# --------------------------------------------------------------------------
class LedWriter:
    """Write pad LED colors via HID Report 0x81."""

    def __init__(self, hidraw_path):
        self._fd = open_hidraw_write(hidraw_path)
        # Report 0x81: 1 byte report ID + 42 bytes data
        self._report = bytearray(43)
        self._report[0] = 0x81

    def set_pad(self, physical_pad, color):
        byte_offset = PAD_PHYSICAL_TO_LED_BYTE.get(physical_pad)
        if byte_offset is not None:
            self._report[byte_offset] = color

    def send(self):
        os.write(self._fd, bytes(self._report))

    def close(self):
        os.close(self._fd)

    def set_t9_layout(self, active_pad=None):
        """Set all pad LEDs to T9 mode colors."""
        for pad in range(1, 17):
            if pad in CHAR_PADS:
                if pad == active_pad:
                    self.set_pad(pad, COLOR_CHAR_ACTIVE)
                else:
                    self.set_pad(pad, COLOR_CHAR_IDLE)
            elif pad == PAD_ENTER:
                self.set_pad(pad, COLOR_ENTER)
            elif pad == PAD_CANCEL:
                self.set_pad(pad, COLOR_CANCEL)
            elif pad == PAD_BACKSPACE:
                self.set_pad(pad, COLOR_BACKSPACE)
            else:
                self.set_pad(pad, COLOR_OFF)  # pad 8
        self.send()

    def all_off(self):
        """Turn off all pad LEDs."""
        for pad in range(1, 17):
            self.set_pad(pad, COLOR_OFF)
        self.send()


# --------------------------------------------------------------------------
# xdotool integration
# --------------------------------------------------------------------------
class XdotoolSearchBridge:
    """Inject text into Mixxx library search via xdotool."""

    def __init__(self, display=":99"):
        self._display = display
        self._env = {**os.environ, "DISPLAY": display}

    def _run(self, *args):
        subprocess.Popen(
            list(args),
            env=self._env,
            stdout=subprocess.DEVNULL,
            stderr=subprocess.DEVNULL,
        )

    def focus_search(self):
        """Focus the Mixxx search bar."""
        self._run("xdotool", "key", "--clearmodifiers", "ctrl+f")

    def type_text(self, text):
        """Clear search and type the full text buffer."""
        # Select all existing text, then type the new text
        self._run("xdotool", "key", "--clearmodifiers", "ctrl+a")
        time.sleep(0.02)
        if text:
            subprocess.Popen(
                ["xdotool", "type", "--clearmodifiers", "--delay", "0", text],
                env=self._env,
                stdout=subprocess.DEVNULL,
                stderr=subprocess.DEVNULL,
            )
        else:
            self._run("xdotool", "key", "--clearmodifiers", "Delete")

    def confirm_search(self):
        """Move focus from search bar to results."""
        self._run("xdotool", "key", "--clearmodifiers", "Tab")

    def cancel_search(self):
        """Clear the search."""
        self._run("xdotool", "key", "--clearmodifiers", "Escape")


# --------------------------------------------------------------------------
# Main daemon
# --------------------------------------------------------------------------
def main():
    display = os.environ.get("DISPLAY", ":99")
    hidraw_path = None

    # Wait for HID device
    while hidraw_path is None:
        hidraw_path = find_mk3_hidraw()
        if hidraw_path is None:
            print("mk3-t9-daemon: waiting for MK3 HID device...", file=sys.stderr)
            time.sleep(3)

    print(f"mk3-t9-daemon: using {hidraw_path}", file=sys.stderr)

    fd_read = open_hidraw_read(hidraw_path)
    leds = LedWriter(hidraw_path)
    xdotool = XdotoolSearchBridge(display)

    t9_active = False
    keyboard_was_pressed = False
    pad_pressed_state = {}

    engine = None  # created on activation

    def on_change(text):
        leds.set_t9_layout(active_pad=engine.get_pending_pad() if engine else None)
        xdotool.type_text(text)

    def on_submit(text):
        nonlocal t9_active
        t9_active = False
        leds.all_off()
        xdotool.confirm_search()
        print(f"mk3-t9-daemon: submitted '{text}'", file=sys.stderr)

    def on_cancel():
        nonlocal t9_active
        t9_active = False
        leds.all_off()
        xdotool.cancel_search()
        print("mk3-t9-daemon: cancelled", file=sys.stderr)

    print("mk3-t9-daemon: listening for keyboard button", file=sys.stderr)

    try:
        while True:
            try:
                data = os.read(fd_read, 64)
            except OSError:
                time.sleep(0.05)
                continue

            if len(data) < 2:
                continue

            report_id = data[0]

            # --- Check keyboard button in Report 0x01 ---
            if report_id == 0x01 and len(data) > KEYBOARD_BYTE:
                pressed = (data[KEYBOARD_BYTE] & KEYBOARD_MASK) != 0

                if pressed and not keyboard_was_pressed:
                    if not t9_active:
                        # Activate T9
                        t9_active = True
                        engine = T9Engine(
                            on_change=on_change,
                            on_submit=on_submit,
                            on_cancel=on_cancel,
                        )
                        pad_pressed_state.clear()
                        leds.set_t9_layout()
                        xdotool.focus_search()
                        print("mk3-t9-daemon: T9 activated", file=sys.stderr)
                    else:
                        # Deactivate T9
                        t9_active = False
                        leds.all_off()
                        engine = None
                        print("mk3-t9-daemon: T9 deactivated", file=sys.stderr)

                keyboard_was_pressed = pressed

            # --- Process pad reports when T9 is active ---
            if t9_active and engine and report_id == PAD_REPORT_ID:
                seen = set()
                for e_idx in range(PAD_MAX_ENTRIES):
                    base = PAD_DATA_START + e_idx * PAD_ENTRY_LENGTH
                    if base + 2 >= len(data):
                        break
                    hw_idx = data[base]
                    if hw_idx >= PAD_COUNT:
                        break
                    pressure = data[base + 1] | (data[base + 2] << 8)
                    phys_pad = PAD_HW_TO_PHYSICAL[hw_idx]

                    seen.add(phys_pad)
                    is_down = pressure >= PAD_PRESSURE_THRESHOLD
                    was_down = pad_pressed_state.get(phys_pad, False)

                    if is_down and not was_down:
                        pad_pressed_state[phys_pad] = True
                        engine.press(phys_pad)
                        # Update LED for active pad
                        leds.set_t9_layout(active_pad=engine.get_pending_pad())
                    elif not is_down and was_down:
                        pad_pressed_state[phys_pad] = False

                # Release pads not in report
                for pad_key in list(pad_pressed_state.keys()):
                    if pad_key not in seen and pad_pressed_state[pad_key]:
                        pad_pressed_state[pad_key] = False

            # --- Tick the engine for commit timeouts ---
            if t9_active and engine:
                if engine.tick():
                    leds.set_t9_layout(active_pad=engine.get_pending_pad())

    except KeyboardInterrupt:
        pass
    finally:
        leds.all_off()
        leds.close()
        os.close(fd_read)
        print("mk3-t9-daemon: shutdown", file=sys.stderr)


if __name__ == "__main__":
    main()
```
